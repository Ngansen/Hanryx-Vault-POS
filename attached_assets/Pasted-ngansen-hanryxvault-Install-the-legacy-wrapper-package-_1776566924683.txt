sudo apt-get update
sudo apt-get install -y xserver-xorg-legacy

sudo chmod u+s /usr/bin/Xorg
sudo chmod u+s /usr/lib/xorg/Xorg

# Confirm both have the 's' bit now
ls -l /usr/bin/Xorg /usr/lib/xorg/Xorg

# Restart kiosk
sudo systemctl restart hanryxvault-kiosk.service
sleep 4
